FROM customer
RIGHT JOIN orders
ON customer.id = orders.customer_id
GROUP BY customer.id;

SELECT firstname,lastname,orderdate,SUM(amount) AS 'total_amount',COUNT(*)
FROM customer
RIGHT JOIN orders
ON customer.id = orders.customer_id
GROUP BY customer.id
ORDER BY total_amount DESC;
